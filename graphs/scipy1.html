In[26]:
# Histogram
plt.figure(figsize=(6,4))
sns.histplot(ordr_amt,bins=10,color="skyblue",edgecolor="black",kde="True",alpha=0.5)

plt.xlabel("total_amount")
plt.ylabel("count")
plt.show()
